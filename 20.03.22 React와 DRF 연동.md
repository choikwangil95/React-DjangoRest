```javascripts
// app.js

import axios from "axios";

class App extends React.Component {
  
  state = {
    post : [],
  };
  
  componentDidMount() {
    try{
      const res = axios.get("http://127.0.0.1:8000/posts/") // api 에서 axios.get으로 data 가져오기
        .then( res =>
          this.setState({ // setState로 state에 data 넣기
            post : res.data
          })
        )
    } catch(e){
      console.log(e);
    }
}

  render() {
  
  const name = "광일";
  const posts = this.state.post;  // posts에 state.post 할당
  console.log(posts);

  return (
      <div>
        {posts? (
          <div>
            {posts.map(post => (
              <div key = {post.id}>
                {post.id} 번 Post : 
                {post.title}
              </div>
            ))}
          </div>
        ) : (
          <div>
            <p>nothing</p>
          </div>
        )}
      </div>
  );
  }
}


export default App;
```
